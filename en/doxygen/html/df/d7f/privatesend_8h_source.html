162  nDenom(nDenom),

165  fReady(fReady),
166  vchSig(std::vector<unsigned char>()),

171 
172  template <typename Stream, typename Operation>
173  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion) {
174  READWRITE(nDenom);
175  READWRITE(vin);
176  READWRITE(nTime);
177  READWRITE(fReady);
178  READWRITE(vchSig);
179  }
180 
188  bool Sign();
190  bool CheckSignature(const CPubKey& pubKeyMasternode);
191 
192  bool Relay(CConnman &connman);
193 
195  bool IsExpired() { return GetAdjustedTime() - nTime > PRIVATESEND_QUEUE_TIMEOUT; }
196 
197  std::string ToString()
198  {
199  return strprintf("nDenom=%d, nTime=%lld, fReady=%s, fTried=%s, masternode=%s",
200  nDenom, nTime, fReady ? "true" : "false", fTried ? "true" : "false", vin.prevout.ToStringShort());
201  }
202 
203  friend bool operator==(const CDarksendQueue& a, const CDarksendQueue& b)
204  {
205  return a.nDenom == b.nDenom && a.vin.prevout == b.vin.prevout && a.nTime == b.nTime && a.fReady == b.fReady;
206  }
