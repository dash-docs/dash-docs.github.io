25 def serialize_script_num(value):
26  r = bytearray(0)
27  if value == 0:
28  return r
29  neg = value < 0
30  absvalue = -value if neg else value
31  while (absvalue):
32  r.append(int(absvalue & 0xff))

34  if r[-1] & 0x80:
35  r.append(0x80 if neg else 0)
36  elif neg:
37  r[-1] |= 0x80
