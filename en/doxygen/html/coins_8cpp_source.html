41 CCoinsMap::iterator CCoinsViewCache::FetchCoin(const COutPoint &outpoint) const {
42  CCoinsMap::iterator it = cacheCoins.find(outpoint);
43  if (it != cacheCoins.end())
44  return it;
45  Coin tmp;
46  if (!base->GetCoin(outpoint, tmp))
47  return cacheCoins.end();
48  CCoinsMap::iterator ret = cacheCoins.emplace(std::piecewise_construct, std::forward_as_tuple(outpoint), std::forward_as_tuple(std::move(tmp))).first;
49  if (ret->second.coin.IsSpent()) {
50  // The parent only has an empty entry for this outpoint; we can consider our
51  // version as fresh.
52  ret->second.flags = CCoinsCacheEntry::FRESH;
53  }
54  cachedCoinsUsage += ret->second.coin.DynamicMemoryUsage();
55  return ret;
56 }

67 void CCoinsViewCache::AddCoin(const COutPoint &outpoint, Coin&& coin, bool possible_overwrite) {
68  assert(!coin.IsSpent());
69  if (coin.out.scriptPubKey.IsUnspendable()) return;
70  CCoinsMap::iterator it;
71  bool inserted;
72  std::tie(it, inserted) = cacheCoins.emplace(std::piecewise_construct, std::forward_as_tuple(outpoint), std::tuple<>());
73  bool fresh = false;
74  if (!inserted) {
75  cachedCoinsUsage -= it->second.coin.DynamicMemoryUsage();
76  }
77  if (!possible_overwrite) {
78  if (!it->second.coin.IsSpent()) {
79  throw std::logic_error("Adding new coin that replaces non-pruned entry");
80  }
81  fresh = !(it->second.flags & CCoinsCacheEntry::DIRTY);
82  }
83  it->second.coin = std::move(coin);
84  it->second.flags |= CCoinsCacheEntry::DIRTY | (fresh ? CCoinsCacheEntry::FRESH : 0);
85  cachedCoinsUsage += it->second.coin.DynamicMemoryUsage();
86 }

88 void AddCoins(CCoinsViewCache& cache, const CTransaction &tx, int nHeight) {
89  bool fCoinbase = tx.IsCoinBase();
90  const uint256& txid = tx.GetHash();
91  for (size_t i = 0; i < tx.vout.size(); ++i) {
92  // Pass fCoinbase as the possible_overwrite flag to AddCoin, in order to correctly
93  // deal with the pre-BIP30 occurrances of duplicate coinbase transactions.
94  cache.AddCoin(COutPoint(txid, i), Coin(tx.vout[i], nHeight, fCoinbase), fCoinbase);
95  }

98 bool CCoinsViewCache::SpendCoin(const COutPoint &outpoint, Coin* moveout) {
99  CCoinsMap::iterator it = FetchCoin(outpoint);
100  if (it == cacheCoins.end()) return false;
101  cachedCoinsUsage -= it->second.coin.DynamicMemoryUsage();
102  if (moveout) {
103  *moveout = std::move(it->second.coin);
104  }
105  if (it->second.flags & CCoinsCacheEntry::FRESH) {
106  cacheCoins.erase(it);
107  } else {
108  it->second.flags |= CCoinsCacheEntry::DIRTY;
109  it->second.coin.Clear();
110  }
111  return true;
112 }

145 bool CCoinsViewCache::BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlockIn) {
146  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
147  if (it->second.flags & CCoinsCacheEntry::DIRTY) { // Ignore non-dirty entries (optimization).
148  CCoinsMap::iterator itUs = cacheCoins.find(it->first);
149  if (itUs == cacheCoins.end()) {
150  // The parent cache does not have an entry, while the child does
151  // We can ignore it if it's both FRESH and pruned in the child
152  if (!(it->second.flags & CCoinsCacheEntry::FRESH && it->second.coin.IsSpent())) {
153  // Otherwise we will need to create it in the parent
154  // and move the data up and mark it as dirty
155  CCoinsCacheEntry& entry = cacheCoins[it->first];
156  entry.coin = std::move(it->second.coin);
157  cachedCoinsUsage += entry.coin.DynamicMemoryUsage();
158  entry.flags = CCoinsCacheEntry::DIRTY;
159  // We can mark it FRESH in the parent if it was FRESH in the child
160  // Otherwise it might have just been flushed from the parent's cache
161  // and already exist in the grandparent
162  if (it->second.flags & CCoinsCacheEntry::FRESH)
163  entry.flags |= CCoinsCacheEntry::FRESH;
164  }
165  } else {
166  // Assert that the child cache entry was not marked FRESH if the
167  // parent cache entry has unspent outputs. If this ever happens,
168  // it means the FRESH flag was misapplied and there is a logic
169  // error in the calling code.
170  if ((it->second.flags & CCoinsCacheEntry::FRESH) && !itUs->second.coin.IsSpent())
171  throw std::logic_error("FRESH flag misapplied to cache entry for base transaction with spendable outputs");
172 
173  // Found the entry in the parent cache
174  if ((itUs->second.flags & CCoinsCacheEntry::FRESH) && it->second.coin.IsSpent()) {
175  // The grandparent does not have an entry, and the child is
176  // modified and being pruned. This means we can just delete
177  // it from the parent.
178  cachedCoinsUsage -= itUs->second.coin.DynamicMemoryUsage();
179  cacheCoins.erase(itUs);
180  } else {
181  // A normal modification.
182  cachedCoinsUsage -= itUs->second.coin.DynamicMemoryUsage();
183  itUs->second.coin = std::move(it->second.coin);
184  cachedCoinsUsage += itUs->second.coin.DynamicMemoryUsage();
185  itUs->second.flags |= CCoinsCacheEntry::DIRTY;
186  // NOTE: It is possible the child has a FRESH flag here in
187  // the event the entry we found in the parent is pruned. But
188  // we must not copy that FRESH flag to the parent as that
189  // pruned state likely still needs to be communicated to the
190  // grandparent.
191  }
192  }
193  }
194  CCoinsMap::iterator itOld = it++;
195  mapCoins.erase(itOld);
196  }
197  hashBlock = hashBlockIn;
198  return true;
199 }

245 double CCoinsViewCache::GetPriority(const CTransaction &tx, int nHeight, CAmount &inChainInputValue) const
246 {
247  inChainInputValue = 0;
248  if (tx.IsCoinBase())
249  return 0.0;
250  double dResult = 0.0;
251  BOOST_FOREACH(const CTxIn& txin, tx.vin)
252  {
253  const Coin& coin = AccessCoin(txin.prevout);
254  if (coin.IsSpent()) continue;
255  if (coin.nHeight <= nHeight) {
256  dResult += coin.out.nValue * (nHeight-coin.nHeight);
257  inChainInputValue += coin.out.nValue;
258  }
259  }
260  return tx.ComputePriority(dResult);
261 }
