132 void TorControlConnection::readcb(struct bufferevent *bev, void *ctx)
133 {
134  TorControlConnection *self = (TorControlConnection*)ctx;
135  struct evbuffer *input = bufferevent_get_input(bev);
136  size_t n_read_out = 0;
137  char *line;
138  assert(input);
139  // If there is not a whole line to read, evbuffer_readln returns NULL
140  while((line = evbuffer_readln(input, &n_read_out, EVBUFFER_EOL_CRLF)) != NULL)
141  {
142  std::string s(line, n_read_out);
143  free(line);
144  if (s.size() < 4) // Short line
145  continue;
146  // <status>(-|+| )<data><CRLF>
147  self->message.code = atoi(s.substr(0,3));
148  self->message.lines.push_back(s.substr(4));
149  char ch = s[3]; // '-','+' or ' '
150  if (ch == ' ') {
151  // Final line, dispatch reply and clean up
152  if (self->message.code >= 600) {
153  // Dispatch async notifications to async handler
154  // Synchronous and asynchronous messages are never interleaved
155  self->async_handler(*self, self->message);
156  } else {
157  if (!self->reply_handlers.empty()) {
158  // Invoke reply handler with message
159  self->reply_handlers.front()(*self, self->message);
160  self->reply_handlers.pop_front();
161  } else {
162  LogPrint("tor", "tor: Received unexpected sync reply %i\n", self->message.code);
163  }
164  }
165  self->message.Clear();
166  }
167  }
168  // Check for size of buffer - protect against memory exhaustion with very long lines
169  // Do this after evbuffer_readln to make sure all full lines have been
170  // removed from the buffer. Everything left is an incomplete line.
171  if (evbuffer_get_length(input) > MAX_LINE_LENGTH) {
172  LogPrintf("tor: Disconnecting because MAX_LINE_LENGTH exceeded\n");
173  self->Disconnect();
174  }

264 static std::map<std::string,std::string> ParseTorReplyMapping(const std::string &s)
265 {
266  std::map<std::string,std::string> mapping;
267  size_t ptr=0;
268  while (ptr < s.size()) {
269  std::string key, value;
270  while (ptr < s.size() && s[ptr] != '=') {
271  key.push_back(s[ptr]);
272  ++ptr;
273  }
274  if (ptr == s.size()) // unexpected end of line
275  return std::map<std::string,std::string>();
276  ++ptr; // skip '='
277  if (ptr < s.size() && s[ptr] == '"') { // Quoted string
278  ++ptr; // skip '='
279  bool escape_next = false;
280  while (ptr < s.size() && (!escape_next && s[ptr] != '"')) {
281  escape_next = (s[ptr] == '\\');
282  value.push_back(s[ptr]);
283  ++ptr;
284  }
285  if (ptr == s.size()) // unexpected end of line
286  return std::map<std::string,std::string>();
287  ++ptr; // skip closing '"'
288  /* TODO: unescape value - according to the spec this depends on the
289  * context, some strings use C-LogPrintf style escape codes, some
290  * don't. So may be better handled at the call site.
291  */
292  } else { // Unquoted value. Note that values can contain '=' at will, just no spaces
293  while (ptr < s.size() && s[ptr] != ' ') {
294  value.push_back(s[ptr]);
295  ++ptr;
296  }
297  }
298  if (ptr < s.size() && s[ptr] == ' ')
299  ++ptr; // skip ' ' after key=value
300  mapping[key] = value;
301  }
302  return mapping;
303 }

496 static std::vector<uint8_t> ComputeResponse(const std::string &key, const std::vector<uint8_t> &cookie, const std::vector<uint8_t> &clientNonce, const std::vector<uint8_t> &serverNonce)
497 {
498  CHMAC_SHA256 computeHash((const uint8_t*)key.data(), key.size());
499  std::vector<uint8_t> computedHash(CHMAC_SHA256::OUTPUT_SIZE, 0);
500  computeHash.Write(begin_ptr(cookie), cookie.size());
501  computeHash.Write(begin_ptr(clientNonce), clientNonce.size());
502  computeHash.Write(begin_ptr(serverNonce), serverNonce.size());
503  computeHash.Finalize(begin_ptr(computedHash));
504  return computedHash;

538 void TorController::protocolinfo_cb(TorControlConnection& conn, const TorControlReply& reply)
539 {
540  if (reply.code == 250) {
541  std::set<std::string> methods;
542  std::string cookiefile;
543  /*
544  * 250-AUTH METHODS=COOKIE,SAFECOOKIE COOKIEFILE="/home/x/.tor/control_auth_cookie"
545  * 250-AUTH METHODS=NULL
546  * 250-AUTH METHODS=HASHEDPASSWORD
547  */
548  BOOST_FOREACH(const std::string &s, reply.lines) {
549  std::pair<std::string,std::string> l = SplitTorReplyLine(s);
550  if (l.first == "AUTH") {
551  std::map<std::string,std::string> m = ParseTorReplyMapping(l.second);
552  std::map<std::string,std::string>::iterator i;
553  if ((i = m.find("METHODS")) != m.end())
554  boost::split(methods, i->second, boost::is_any_of(","));
555  if ((i = m.find("COOKIEFILE")) != m.end())
556  cookiefile = i->second;
557  } else if (l.first == "VERSION") {
558  std::map<std::string,std::string> m = ParseTorReplyMapping(l.second);
559  std::map<std::string,std::string>::iterator i;
560  if ((i = m.find("Tor")) != m.end()) {
561  LogPrint("tor", "tor: Connected to Tor version %s\n", i->second);
562  }
563  }
564  }
565  BOOST_FOREACH(const std::string &s, methods) {
566  LogPrint("tor", "tor: Supported authentication method: %s\n", s);
567  }
568  // Prefer NULL, otherwise SAFECOOKIE. If a password is provided, use HASHEDPASSWORD
569  /* Authentication:
570  * cookie: hex-encoded ~/.tor/control_auth_cookie
571  * password: "password"
572  */
573  std::string torpassword = GetArg("-torpassword", "");
574  if (methods.count("NULL")) {
575  LogPrint("tor", "tor: Using NULL authentication\n");
576  conn.Command("AUTHENTICATE", boost::bind(&TorController::auth_cb, this, _1, _2));
577  } else if (methods.count("SAFECOOKIE")) {
578  // Cookie: hexdump -e '32/1 "%02x""\n"' ~/.tor/control_auth_cookie
579  LogPrint("tor", "tor: Using SAFECOOKIE authentication, reading cookie authentication from %s\n", cookiefile);
580  std::pair<bool,std::string> status_cookie = ReadBinaryFile(cookiefile, TOR_COOKIE_SIZE);
581  if (status_cookie.first && status_cookie.second.size() == TOR_COOKIE_SIZE) {
582  // conn.Command("AUTHENTICATE " + HexStr(status_cookie.second), boost::bind(&TorController::auth_cb, this, _1, _2));
583  cookie = std::vector<uint8_t>(status_cookie.second.begin(), status_cookie.second.end());
584  clientNonce = std::vector<uint8_t>(TOR_NONCE_SIZE, 0);
585  GetRandBytes(&clientNonce[0], TOR_NONCE_SIZE);
586  conn.Command("AUTHCHALLENGE SAFECOOKIE " + HexStr(clientNonce), boost::bind(&TorController::authchallenge_cb, this, _1, _2));
587  } else {
588  if (status_cookie.first) {
589  LogPrintf("tor: Authentication cookie %s is not exactly %i bytes, as is required by the spec\n", cookiefile, TOR_COOKIE_SIZE);
590  } else {
591  LogPrintf("tor: Authentication cookie %s could not be opened (check permissions)\n", cookiefile);
592  }
593  }
594  } else if (methods.count("HASHEDPASSWORD")) {
595  if (!torpassword.empty()) {
596  LogPrint("tor", "tor: Using HASHEDPASSWORD authentication\n");
597  boost::replace_all(torpassword, "\"", "\\\"");
598  conn.Command("AUTHENTICATE \"" + torpassword + "\"", boost::bind(&TorController::auth_cb, this, _1, _2));
599  } else {
600  LogPrintf("tor: Password authentication required, but no password provided with -torpassword\n");
601  }
602  } else {
603  LogPrintf("tor: No supported authentication method\n");
604  }
605  } else {
606  LogPrintf("tor: Requesting protocol info failed\n");
607  }
