238 struct CAddressIndexKey {
239  unsigned int type;
240  uint160 hashBytes;
241  int blockHeight;
242  unsigned int txindex;
243  uint256 txhash;
244  size_t index;
245  bool spending;
246 
247  size_t GetSerializeSize(int nType, int nVersion) const {
248  return 66;
249  }
250  template<typename Stream>
251  void Serialize(Stream& s, int nType, int nVersion) const {
252  ser_writedata8(s, type);
253  hashBytes.Serialize(s, nType, nVersion);
254  // Heights are stored big-endian for key sorting in LevelDB
255  ser_writedata32be(s, blockHeight);
256  ser_writedata32be(s, txindex);
257  txhash.Serialize(s, nType, nVersion);
258  ser_writedata32(s, index);
259  char f = spending;
260  ser_writedata8(s, f);
261  }
262  template<typename Stream>
263  void Unserialize(Stream& s, int nType, int nVersion) {
264  type = ser_readdata8(s);
265  hashBytes.Unserialize(s, nType, nVersion);
266  blockHeight = ser_readdata32be(s);
267  txindex = ser_readdata32be(s);
268  txhash.Unserialize(s, nType, nVersion);
269  index = ser_readdata32(s);
270  char f = ser_readdata8(s);
271  spending = f;
272  }
