79 bool CCoinsViewDB::BatchWrite(CCoinsMap &mapCoins, const uint256 &hashBlock) {
80  CDBBatch batch(db);
81  size_t count = 0;
82  size_t changed = 0;
83  for (CCoinsMap::iterator it = mapCoins.begin(); it != mapCoins.end();) {
84  if (it->second.flags & CCoinsCacheEntry::DIRTY) {
85  CoinEntry entry(&it->first);
86  if (it->second.coin.IsSpent())
87  batch.Erase(entry);
88  else
89  batch.Write(entry, it->second.coin);
90  changed++;
91  }
92  count++;
93  CCoinsMap::iterator itOld = it++;
94  mapCoins.erase(itOld);
95  }
96  if (!hashBlock.IsNull())
97  batch.Write(DB_BEST_BLOCK, hashBlock);
98 
99  bool ret = db.WriteBatch(batch);
100  LogPrint("coindb", "Committed %u changed transaction outputs (out of %u) to coin database...\n", (unsigned int)changed, (unsigned int)count);
101  return ret;
102 }

408  CCoins() : fCoinBase(false), vout(0), nHeight(0) { }

410  template<typename Stream>
411  void Unserialize(Stream &s, int nType, int nVersion) {
412  unsigned int nCode = 0;
413  // version
414  int nVersionDummy;
415  ::Unserialize(s, VARINT(nVersionDummy), nType, nVersion);
416  // header code
417  ::Unserialize(s, VARINT(nCode), nType, nVersion);
418  fCoinBase = nCode & 1;
419  std::vector<bool> vAvail(2, false);
420  vAvail[0] = (nCode & 2) != 0;
421  vAvail[1] = (nCode & 4) != 0;
422  unsigned int nMaskCode = (nCode / 8) + ((nCode & 6) != 0 ? 0 : 1);
423  // spentness bitmask
424  while (nMaskCode > 0) {
425  unsigned char chAvail = 0;
426  ::Unserialize(s, chAvail, nType, nVersion);
427  for (unsigned int p = 0; p < 8; p++) {
428  bool f = (chAvail & (1 << p)) != 0;
429  vAvail.push_back(f);
430  }
431  if (chAvail != 0)
432  nMaskCode--;
433  }
434  // txouts themself
435  vout.assign(vAvail.size(), CTxOut());
436  for (unsigned int i = 0; i < vAvail.size(); i++) {
437  if (vAvail[i])
438  ::Unserialize(s, REF(CTxOutCompressor(vout[i])), nType, nVersion);
439  }
440  // coinbase height
441  ::Unserialize(s, VARINT(nHeight), nType, nVersion);
442  }
443 };
444

451 bool CCoinsViewDB::Upgrade() {
452  std::unique_ptr<CDBIterator> pcursor(db.NewIterator());
453  pcursor->Seek(std::make_pair(DB_COINS, uint256()));
454  if (!pcursor->Valid()) {
455  return true;
456  }
457 
458  int64_t count = 0;
459  LogPrintf("Upgrading utxo-set database...\n");
460  LogPrintf("[0%%]...");
461  size_t batch_size = 1 << 24;
462  CDBBatch batch(db);
463  uiInterface.SetProgressBreakAction(StartShutdown);
464  int reportDone = 0;
465  std::pair<unsigned char, uint256> key;
466  std::pair<unsigned char, uint256> prev_key = {DB_COINS, uint256()};
467  while (pcursor->Valid()) {
468  boost::this_thread::interruption_point();
469  if (ShutdownRequested()) {
470  break;
471  }
472  if (pcursor->GetKey(key) && key.first == DB_COINS) {
473  if (count++ % 256 == 0) {
474  uint32_t high = 0x100 * *key.second.begin() + *(key.second.begin() + 1);
475  int percentageDone = (int)(high * 100.0 / 65536.0 + 0.5);
476  uiInterface.ShowProgress(_("Upgrading UTXO database") + "\n"+ _("(press q to shutdown and continue later)") + "\n", percentageDone);
477  if (reportDone < percentageDone/10) {
478  // report max. every 10% step
479  LogPrintf("[%d%%]...", percentageDone);
480  reportDone = percentageDone/10;
481  }
482  }
483  CCoins old_coins;
484  if (!pcursor->GetValue(old_coins)) {
485  return error("%s: cannot parse CCoins record", __func__);
486  }
487  COutPoint outpoint(key.second, 0);
488  for (size_t i = 0; i < old_coins.vout.size(); ++i) {
489  if (!old_coins.vout[i].IsNull() && !old_coins.vout[i].scriptPubKey.IsUnspendable()) {
490  Coin newcoin(std::move(old_coins.vout[i]), old_coins.nHeight, old_coins.fCoinBase);
491  outpoint.n = i;
492  CoinEntry entry(&outpoint);
493  batch.Write(entry, newcoin);
494  }
495  }
496  batch.Erase(key);
497  if (batch.SizeEstimate() > batch_size) {
498  db.WriteBatch(batch);
499  batch.Clear();
500  db.CompactRange(prev_key, key);
501  prev_key = key;
502  }
503  pcursor->Next();
504  } else {
505  break;
506  }
507  }
508  db.WriteBatch(batch);
509  db.CompactRange({DB_COINS, uint256()}, key);
510  uiInterface.SetProgressBreakAction(std::function<void(void)>());
511  LogPrintf("[%s].\n", ShutdownRequested() ? "CANCELLED" : "DONE");
512  return !ShutdownRequested();
513 }
