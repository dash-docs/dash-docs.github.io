31 void NetworkStyle::rotateColor(QColor& col, const int iconColorHueShift, const int iconColorSaturationReduction)
32 {
33  int h, s, l, a;
34  col.getHsl(&h, &s, &l, &a);
35 
36  // rotate color on RGB color circle
37  h += iconColorHueShift;
38  // change saturation value
39  s -= iconColorSaturationReduction;
40  s = std::max(s, 0);
41 
42  col.setHsl(h,s,l,a);
