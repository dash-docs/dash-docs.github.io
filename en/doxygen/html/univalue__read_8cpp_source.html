18 // convert hexadecimal string to unsigned integer
19 static const char *hatoui(const char *first, const char *last,
20  unsigned int& out)
21 {
22  unsigned int result = 0;
23  for (; first != last; ++first)
24  {
25  int digit;
26  if (json_isdigit(*first))
27  digit = *first - '0';
28 
29  else if (*first >= 'a' && *first <= 'f')
30  digit = *first - 'a' + 10;
31 
32  else if (*first >= 'A' && *first <= 'F')
33  digit = *first - 'A' + 10;
34 
35  else
36  break;
37 
38  result = 16 * result + digit;
39  }
40  out = result;
41 
42  return first;
43 }

46  const char *raw)
47 {
48  tokenVal.clear();
49  consumed = 0;
50 
51  const char *rawStart = raw;
52 
53  while ((*raw) && (json_isspace(*raw))) // skip whitespace
54  raw++;
55 
56  switch (*raw) {
57 
58  case 0:
59  return JTOK_NONE;
60 
61  case '{':
62  raw++;
63  consumed = (raw - rawStart);
64  return JTOK_OBJ_OPEN;
65  case '}':
66  raw++;
67  consumed = (raw - rawStart);
68  return JTOK_OBJ_CLOSE;
69  case '[':
70  raw++;
71  consumed = (raw - rawStart);
72  return JTOK_ARR_OPEN;
73  case ']':
74  raw++;
75  consumed = (raw - rawStart);
76  return JTOK_ARR_CLOSE;
77 
78  case ':':
79  raw++;
80  consumed = (raw - rawStart);
81  return JTOK_COLON;
82  case ',':
83  raw++;
84  consumed = (raw - rawStart);
85  return JTOK_COMMA;
86 
87  case 'n':
88  case 't':
89  case 'f':
90  if (!strncmp(raw, "null", 4)) {
91  raw += 4;
92  consumed = (raw - rawStart);
93  return JTOK_KW_NULL;
94  } else if (!strncmp(raw, "true", 4)) {
95  raw += 4;
96  consumed = (raw - rawStart);
97  return JTOK_KW_TRUE;
98  } else if (!strncmp(raw, "false", 5)) {
99  raw += 5;
100  consumed = (raw - rawStart);
101  return JTOK_KW_FALSE;
102  } else
103  return JTOK_ERR;
104 
105  case '-':
106  case '0':
107  case '1':
108  case '2':
109  case '3':
110  case '4':
111  case '5':
112  case '6':
113  case '7':
114  case '8':
115  case '9': {
116  // part 1: int
117  string numStr;
118 
119  const char *first = raw;
120 
121  const char *firstDigit = first;
122  if (!json_isdigit(*firstDigit))
123  firstDigit++;
124  if ((*firstDigit == '0') && json_isdigit(firstDigit[1]))
125  return JTOK_ERR;
126 
127  numStr += *raw; // copy first char
128  raw++;
129 
130  if ((*first == '-') && (!json_isdigit(*raw)))
131  return JTOK_ERR;
132 
133  while ((*raw) && json_isdigit(*raw)) { // copy digits
134  numStr += *raw;
135  raw++;
136  }
137 
138  // part 2: frac
139  if (*raw == '.') {
140  numStr += *raw; // copy .
141  raw++;
142 
143  if (!json_isdigit(*raw))
144  return JTOK_ERR;
145  while ((*raw) && json_isdigit(*raw)) { // copy digits
146  numStr += *raw;
147  raw++;
148  }
149  }
150 
151  // part 3: exp
152  if (*raw == 'e' || *raw == 'E') {
153  numStr += *raw; // copy E
154  raw++;
155 
156  if (*raw == '-' || *raw == '+') { // copy +/-
157  numStr += *raw;
158  raw++;
159  }
160 
161  if (!json_isdigit(*raw))
162  return JTOK_ERR;
163  while ((*raw) && json_isdigit(*raw)) { // copy digits
164  numStr += *raw;
165  raw++;
166  }
167  }
168 
169  tokenVal = numStr;
170  consumed = (raw - rawStart);
171  return JTOK_NUMBER;
172  }
173 
174  case '"': {
175  raw++; // skip "
176 
177  string valStr;
178  JSONUTF8StringFilter writer(valStr);
179 
180  while (*raw) {
181  if ((unsigned char)*raw < 0x20)
182  return JTOK_ERR;
183 
184  else if (*raw == '\\') {
185  raw++; // skip backslash
186 
187  switch (*raw) {
188  case '"': writer.push_back('\"'); break;
189  case '\\': writer.push_back('\\'); break;
190  case '/': writer.push_back('/'); break;
191  case 'b': writer.push_back('\b'); break;
192  case 'f': writer.push_back('\f'); break;
193  case 'n': writer.push_back('\n'); break;
194  case 'r': writer.push_back('\r'); break;
195  case 't': writer.push_back('\t'); break;
196 
197  case 'u': {
198  unsigned int codepoint;
199  if (hatoui(raw + 1, raw + 1 + 4, codepoint) !=
200  raw + 1 + 4)
201  return JTOK_ERR;
202  writer.push_back_u(codepoint);
203  raw += 4;
204  break;
205  }
206  default:
207  return JTOK_ERR;
208 
209  }
210 
211  raw++; // skip esc'd char
212  }
213 
214  else if (*raw == '"') {
215  raw++; // skip "
216  break; // stop scanning
217  }
218 
219  else {
220  writer.push_back(*raw);
221  raw++;
222  }
223  }
224 
225  if (!writer.finalize())
226  return JTOK_ERR;
227  tokenVal = valStr;
228  consumed = (raw - rawStart);
229  return JTOK_STRING;
230  }
231 
232  default:
233  return JTOK_ERR;
234  }
235 }

259  enum jtokentype last_tok = JTOK_NONE;
260  do {
261  last_tok = tok;
262 
263  tok = getJsonToken(tokenVal, consumed, raw);
264  if (tok == JTOK_NONE || tok == JTOK_ERR)
265  return false;

267

269  tok == JTOK_OBJ_OPEN || tok == JTOK_ARR_OPEN;
270

272  if (!isValueOpen)
273  return false;
274  clearExpect(VALUE);
275 
276  } else if (expect(ARR_VALUE)) {
277  bool isArrValue = isValueOpen || (tok == JTOK_ARR_CLOSE);
278  if (!isArrValue)
279  return false;

281  clearExpect(ARR_VALUE);
282 
283  } else if (expect(OBJ_NAME)) {
284  bool isObjName = (tok == JTOK_OBJ_CLOSE || tok == JTOK_STRING);
285  if (!isObjName)
286  return false;
287 
288  } else if (expect(COLON)) {
289  if (tok != JTOK_COLON)
290  return false;
291  clearExpect(COLON);
292 
293  } else if (!expect(COLON) && (tok == JTOK_COLON)) {
294  return false;
295  }
296 
297  if (expect(NOT_VALUE)) {
298  if (isValueOpen)
299  return false;
300  clearExpect(NOT_VALUE);

302

304

306  case JTOK_ARR_OPEN: {

309  if (utyp == VOBJ)

311  else

314  } else {

317  top->values.push_back(tmpVal);
318

320  stack.push_back(newTop);
321  }
322 
323  if (utyp == VOBJ)
324  setExpect(OBJ_NAME);
325  else
326  setExpect(ARR_VALUE);
327  break;
328  }
329 
330  case JTOK_OBJ_CLOSE:
331  case JTOK_ARR_CLOSE: {
332  if (!stack.size() || (last_tok == JTOK_COMMA))
333  return false;
334 
335  VType utyp = (tok == JTOK_OBJ_CLOSE ? VOBJ : VARR);
336  UniValue *top = stack.back();
337  if (utyp != top->getType())
338  return false;
339 
340  stack.pop_back();

342  setExpect(NOT_VALUE);
343  break;
344  }
345 
346  case JTOK_COLON: {
347  if (!stack.size())
348  return false;
349 
350  UniValue *top = stack.back();
351  if (top->getType() != VOBJ)
352  return false;
353 
354  setExpect(VALUE);
355  break;
356  }
357 
358  case JTOK_COMMA: {
359  if (!stack.size() ||
360  (last_tok == JTOK_COMMA) || (last_tok == JTOK_ARR_OPEN))
361  return false;
362 
363  UniValue *top = stack.back();
364  if (top->getType() == VOBJ)
365  setExpect(OBJ_NAME);
366  else
367  setExpect(ARR_VALUE);
368  break;
369  }
370 
371  case JTOK_KW_NULL:
372  case JTOK_KW_TRUE:
373  case JTOK_KW_FALSE: {
374  if (!stack.size())
375  return false;
376 
377  UniValue tmpVal;
378  switch (tok) {
379  case JTOK_KW_NULL:
380  // do nothing more
381  break;
382  case JTOK_KW_TRUE:
383  tmpVal.setBool(true);
384  break;
385  case JTOK_KW_FALSE:
386  tmpVal.setBool(false);
387  break;
388  default: /* impossible */ break;
389  }
390 
391  UniValue *top = stack.back();
392  top->values.push_back(tmpVal);
393 
394  setExpect(NOT_VALUE);
395  break;
396  }
397 
398  case JTOK_NUMBER: {
399  if (!stack.size())
400  return false;
401 
402  UniValue tmpVal(VNUM, tokenVal);
403  UniValue *top = stack.back();
404  top->values.push_back(tmpVal);
405 
406  setExpect(NOT_VALUE);
407  break;
408  }
409 
410  case JTOK_STRING: {
411  if (!stack.size())
412  return false;
413 
414  UniValue *top = stack.back();
415 
416  if (expect(OBJ_NAME)) {
417  top->keys.push_back(tokenVal);
418  clearExpect(OBJ_NAME);
419  setExpect(COLON);
420  } else {
421  UniValue tmpVal(VSTR, tokenVal);
422  top->values.push_back(tmpVal);
423  }
424 
425  setExpect(NOT_VALUE);
426  break;
427  }
428 
429  default:
430  return false;
431  }
432  } while (!stack.empty ());
433 
434  /* Check that nothing follows the initial construct (parsed above). */
435  tok = getJsonToken(tokenVal, consumed, raw);
436  if (tok != JTOK_NONE)
437  return false;
438 
439  return true;
440 }
