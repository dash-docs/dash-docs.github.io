36 double GetDifficulty(const CBlockIndex* blockindex)
37 {
38  // Floating point number that is a multiple of the minimum difficulty,
39  // minimum difficulty = 1.0.
40  if (blockindex == NULL)
41  {
42  if (chainActive.Tip() == NULL)
43  return 1.0;
44  else
45  blockindex = chainActive.Tip();
46  }
47 
48  int nShift = (blockindex->nBits >> 24) & 0xff;
49 
50  double dDiff =
51  (double)0x0000ffff / (double)(blockindex->nBits & 0x00ffffff);
52 
53  while (nShift < 29)
54  {
55  dDiff *= 256.0;
56  nShift++;
57  }
58  while (nShift > 29)
59  {
60  dDiff /= 256.0;
61  nShift--;
62  }
63 
64  return dDiff;
65 }

570 static void ApplyStats(CCoinsStats &stats, CHashWriter& ss, const uint256& hash, const std::map<uint32_t, Coin>& outputs)
571 {
572  assert(!outputs.empty());
573  ss << hash;
574  ss << VARINT(outputs.begin()->second.nHeight * 2 + outputs.begin()->second.fCoinBase);
575  stats.nTransactions++;
576  for (const auto output : outputs) {
577  ss << VARINT(output.first + 1);
578  ss << *(const CScriptBase*)(&output.second.out.scriptPubKey);
579  ss << VARINT(output.second.out.nValue);
580  stats.nTransactionOutputs++;
581  stats.nTotalAmount += output.second.out.nValue;
582  }
583  ss << VARINT(0);
584 }

587 static bool GetUTXOStats(CCoinsView *view, CCoinsStats &stats)
588 {
589  boost::scoped_ptr<CCoinsViewCursor> pcursor(view->Cursor());
590 
591  CHashWriter ss(SER_GETHASH, PROTOCOL_VERSION);
592  stats.hashBlock = pcursor->GetBestBlock();

594  LOCK(cs_main);

599  std::map<uint32_t, Coin> outputs;

615  }
