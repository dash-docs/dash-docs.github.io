155 void OptionsDialog::setModel(OptionsModel *model)
156 {
157  this->model = model;
158 
159  if(model)
160  {
161  /* check if client restart is needed and show persistent message */
162  if (model->isRestartRequired())
163  showRestartWarning(true);
164 
165  QString strLabel = model->getOverriddenByCommandLine();
166  if (strLabel.isEmpty())
167  strLabel = tr("none");
168  ui->overriddenByCommandLineLabel->setText(strLabel);
169 
170  mapper->setModel(model);
171  setMapper();
172  mapper->toFirst();
173 
174  updateDefaultProxyNets();
175  }
176 
177  /* warn when one of the following settings changes by user action (placed here so init via mapper doesn't trigger them) */
178 
179  /* Main */
180  connect(ui->databaseCache, SIGNAL(valueChanged(int)), this, SLOT(showRestartWarning()));
181  connect(ui->threadsScriptVerif, SIGNAL(valueChanged(int)), this, SLOT(showRestartWarning()));
182  /* Wallet */
183  connect(ui->showMasternodesTab, SIGNAL(clicked(bool)), this, SLOT(showRestartWarning()));
184  connect(ui->spendZeroConfChange, SIGNAL(clicked(bool)), this, SLOT(showRestartWarning()));
185  /* Network */
186  connect(ui->allowIncoming, SIGNAL(clicked(bool)), this, SLOT(showRestartWarning()));
187  connect(ui->connectSocks, SIGNAL(clicked(bool)), this, SLOT(showRestartWarning()));
188  connect(ui->connectSocksTor, SIGNAL(clicked(bool)), this, SLOT(showRestartWarning()));
189  /* Display */
190  connect(ui->digits, SIGNAL(valueChanged()), this, SLOT(showRestartWarning()));
191  connect(ui->theme, SIGNAL(valueChanged()), this, SLOT(showRestartWarning()));
192  connect(ui->lang, SIGNAL(valueChanged()), this, SLOT(showRestartWarning()));

316 void OptionsDialog::updateProxyValidationState()
317 {
318  QValidatedLineEdit *pUiProxyIp = ui->proxyIp;
319  QValidatedLineEdit *otherProxyWidget = (pUiProxyIp == ui->proxyIpTor) ? ui->proxyIp : ui->proxyIpTor;
320  if (pUiProxyIp->isValid() && (!ui->proxyPort->isEnabled() || ui->proxyPort->text().toInt() > 0) && (!ui->proxyPortTor->isEnabled() || ui->proxyPortTor->text().toInt() > 0))
321  {
322  setOkButtonState(otherProxyWidget->isValid()); //only enable ok button if both proxys are valid
323  ui->statusLabel->clear();
324  }
325  else
326  {
327  setOkButtonState(false);
328  ui->statusLabel->setStyleSheet("QLabel { color: red; }");
329  ui->statusLabel->setText(tr("The supplied proxy address is invalid."));
330  }
