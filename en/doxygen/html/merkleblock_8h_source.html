93  READWRITE(vBytes);
94  CPartialMerkleTree &us = *(const_cast<CPartialMerkleTree*>(this));
95  us.vBits.resize(vBytes.size() * 8);
96  for (unsigned int p = 0; p < us.vBits.size(); p++)
97  us.vBits[p] = (vBytes[p / 8] & (1 << (p % 8))) != 0;
