556  template <typename Stream, typename Operation>
557  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion) {
558  if (!(nType & SER_GETHASH))
559  READWRITE(nVersion);
561  READWRITE(nCreditDebit);
562  READWRITE(nTime);
563  READWRITE(LIMITED_STRING(strOtherAccount, 65536));
564 
565  if (!ser_action.ForRead())
566  {
567  WriteOrderPos(nOrderPos, mapValue);
568 
569  if (!(mapValue.empty() && _ssExtra.empty()))
570  {
571  CDataStream ss(nType, nVersion);
572  ss.insert(ss.begin(), '\0');
573  ss << mapValue;
574  ss.insert(ss.end(), _ssExtra.begin(), _ssExtra.end());
575  strComment.append(ss.str());
576  }
577  }
578 
579  READWRITE(LIMITED_STRING(strComment, 65536));
580 
581  size_t nSepPos = strComment.find("\0", 0, 1);
582  if (ser_action.ForRead())
583  {
584  mapValue.clear();
585  if (std::string::npos != nSepPos)
586  {
587  CDataStream ss(std::vector<char>(strComment.begin() + nSepPos + 1, strComment.end()), nType, nVersion);
588  ss >> mapValue;
589  _ssExtra = std::vector<char>(ss.begin(), ss.end());
590  }
591  ReadOrderPos(nOrderPos, mapValue);
592  }
593  if (std::string::npos != nSepPos)
594  strComment.erase(nSepPos);
595 
596  mapValue.erase("n");
597  }
598 
599 private:
600  std::vector<char> _ssExtra;
