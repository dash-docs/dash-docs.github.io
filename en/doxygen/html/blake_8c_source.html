838 static void
839 blake32_close(sph_blake_small_context *sc,
840  unsigned ub, unsigned n, void *dst, size_t out_size_w32)
841 {
842  union {
843  unsigned char buf[64];
844  sph_u32 dummy;
845  } u;
846  size_t ptr, k;
847  unsigned bit_len;
848  unsigned z;
849  sph_u32 th, tl;
850  unsigned char *out;
851 
852  ptr = sc->ptr;
853  bit_len = ((unsigned)ptr << 3) + n;
854  z = 0x80 >> n;
855  u.buf[ptr] = ((ub & -z) | z) & 0xFF;
856  tl = sc->T0 + bit_len;
857  th = sc->T1;
858  if (ptr == 0 && n == 0) {
859  sc->T0 = SPH_C32(0xFFFFFE00);
860  sc->T1 = SPH_C32(0xFFFFFFFF);
861  } else if (sc->T0 == 0) {
862  sc->T0 = SPH_C32(0xFFFFFE00) + bit_len;
863  sc->T1 = SPH_T32(sc->T1 - 1);
864  } else {
865  sc->T0 -= 512 - bit_len;
866  }
867  if (bit_len <= 446) {
868  memset(u.buf + ptr + 1, 0, 55 - ptr);
869  if (out_size_w32 == 8)
870  u.buf[55] |= 1;
871  sph_enc32be_aligned(u.buf + 56, th);
872  sph_enc32be_aligned(u.buf + 60, tl);
873  blake32(sc, u.buf + ptr, 64 - ptr);
874  } else {
875  memset(u.buf + ptr + 1, 0, 63 - ptr);
876  blake32(sc, u.buf + ptr, 64 - ptr);
877  sc->T0 = SPH_C32(0xFFFFFE00);
878  sc->T1 = SPH_C32(0xFFFFFFFF);
879  memset(u.buf, 0, 56);
