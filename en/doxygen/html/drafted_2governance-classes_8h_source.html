70  * CLASS INHERITANCE

94  * -> DAO

98  * -> CONTRACT1 (INTERNAL CONTRACT)
99  * -> PROJECT1
100  * -> PROJECT1
101  * -> CONTRACT2 (EXTERNAL CONTRACT)
102  * -> PROPOSAL (GENERIC FUNDING)

104  * -> REPORT1
