132 unsigned int CTransaction::CalculateModifiedSize(unsigned int nTxSize) const
133 {
134  // In order to avoid disincentivizing cleaning up the UTXO set we don't count
135  // the constant overhead for each txin and up to 110 bytes of scriptSig (which
136  // is enough to cover a compressed pubkey p2sh redemption) for priority.
137  // Providing any more cleanup incentive than making additional inputs free would
138  // risk encouraging people to create junk outputs to redeem later.
139  if (nTxSize == 0)
140  nTxSize = ::GetSerializeSize(*this, SER_NETWORK, PROTOCOL_VERSION);
141  for (std::vector<CTxIn>::const_iterator it(vin.begin()); it != vin.end(); ++it)
142  {
143  unsigned int offset = 41U + std::min(110U, (unsigned int)it->scriptSig.size());
144  if (nTxSize > offset)
145  nTxSize -= offset;
146  }
147  return nTxSize;
148 }
