154 void SyncWithValidationInterfaceQueue() {
155  AssertLockNotHeld(cs_main);
156  // Block until the validation queue drains
157  std::promise<void> promise;
158  CallFunctionInValidationInterfaceQueue([&promise] {
159  promise.set_value();
160  });
161  promise.get_future().wait();
162 }
