349  explicit CDiskBlockIndex(const CBlockIndex* pindex) : CBlockIndex(*pindex) {
350  hash = (hash == uint256() ? pindex->GetBlockHash() : hash);
351  hashPrev = (pprev ? pprev->GetBlockHash() : uint256());
352  }
353 
354  ADD_SERIALIZE_METHODS;
355 
356  template <typename Stream, typename Operation>
357  inline void SerializationOp(Stream& s, Operation ser_action, int nType, int nVersion) {
358  if (!(nType & SER_GETHASH))
359  READWRITE(VARINT(nVersion));
360 
361  READWRITE(VARINT(nHeight));
362  READWRITE(VARINT(nStatus));
363  READWRITE(VARINT(nTx));
364  if (nStatus & (BLOCK_HAVE_DATA | BLOCK_HAVE_UNDO))
365  READWRITE(VARINT(nFile));
366  if (nStatus & BLOCK_HAVE_DATA)
367  READWRITE(VARINT(nDataPos));
368  if (nStatus & BLOCK_HAVE_UNDO)
369  READWRITE(VARINT(nUndoPos));
370 
371  // block hash
372  READWRITE(hash);
373  // block header
374  READWRITE(this->nVersion);
375  READWRITE(hashPrev);
376  READWRITE(hashMerkleRoot);
377  READWRITE(nTime);
378  READWRITE(nBits);
379  READWRITE(nNonce);
380  }
381 
382  uint256 GetBlockHash() const
383  {
384  if(hash != uint256()) return hash;
385  // should never really get here, keeping this as a fallback
386  CBlockHeader block;
387  block.nVersion = nVersion;
388  block.hashPrevBlock = hashPrev;
389  block.hashMerkleRoot = hashMerkleRoot;
390  block.nTime = nTime;
391  block.nBits = nBits;
392  block.nNonce = nNonce;
393  return block.GetHash();
394  }
395 
396 
397  std::string ToString() const
398  {
399  std::string str = "CDiskBlockIndex(";
400  str += CBlockIndex::ToString();
401  str += strprintf("\n hashBlock=%s, hashPrev=%s)",
402  GetBlockHash().ToString(),
