41 static void from_hex(unsigned char* data, int len, const char* hex) {
42  int p;
43  for (p = 0; p < len; p++) {
44  int v = 0;
45  int n;
46  for (n = 0; n < 2; n++) {
47  assert((*hex >= '0' && *hex <= '9') || (*hex >= 'a' && *hex <= 'f'));
48  if (*hex >= '0' && *hex <= '9') {
49  v |= (*hex - '0') << (4 * (1 - n));
50  } else {
51  v |= (*hex - 'a' + 10) << (4 * (1 - n));
52  }
53  hex++;
54  }
55  *(data++) = v;
56  }
57  assert(*hex == 0);
58 }
