26 void NetworkStyle::rotateColors(QImage& img, const int iconColorHueShift, const int iconColorSaturationReduction) {
27  int h,s,l,a;
28 
29  // traverse though lines
30  for(int y=0;y<img.height();y++)
31  {
32  QRgb *scL = reinterpret_cast< QRgb *>( img.scanLine( y ) );
33 
34  // loop through pixels
35  for(int x=0;x<img.width();x++)
36  {
37  // preserve alpha because QColor::getHsl doesen't return the alpha value
38  a = qAlpha(scL[x]);
39  QColor col(scL[x]);
40 
41  // get hue value
42  col.getHsl(&h,&s,&l);
43 
44  // rotate color on RGB color circle
45  // 70Â° should end up with the typical "testnet" green (in bitcoin)
46  h+=iconColorHueShift;
47 
48  // change saturation value
49  s -= iconColorSaturationReduction;
50  s = std::max(s, 0);
51 
52  col.setHsl(h,s,l,a);
53 
54  // set the pixel
55  scL[x] = col.rgba();
56  }
57  }
58 }
