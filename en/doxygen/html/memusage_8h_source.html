46 static inline size_t MallocUsage(size_t alloc)
47 {
48  // Measured on libc6 2.19 on Linux.
49  if (alloc == 0) {
50  return 0;
51  } else if (sizeof(void*) == 8) {
52  return ((alloc + 31) >> 4) << 4;
53  } else if (sizeof(void*) == 4) {
54  return ((alloc + 15) >> 3) << 3;
55  } else {
56  assert(0);
57  }
58 }
